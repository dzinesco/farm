# 3D View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement a high-quality 3D architectural view for FRMX that bidirectionally syncs with the existing 2D floor plan, rendering walls as proper extruded 3D meshes with realistic materials, camera controls, and interactive selection.

**Architecture:** The 3D scene consumes the existing Zustand store directly. Wall centerlines are extruded into 3D meshes using the geometry package's `WallFrame` system. React Three Fiber wraps Three.js with React reconciler; Drei provides OrbitControls, Environment, and helpers. LEVA provides debug toggles. Each wall becomes a `Wall3D` component that reads from the store and dispatches updates.

**Tech Stack:** @react-three/fiber 8.x, @react-three/drei 9.x, three.js 0.172, leva, Zustand 5 + Immer, TypeScript strict mode, React 19.

```typescript
// apps/web/src/features/three/utils/wallToGeometry.ts
import type { Point2D } from '@frmx/core'

export interface WallSegment {
  start: Point2D
  end: Point2D
  length: number
  direction: Point2D  // normalized
  normal: Point2D     // left-hand perpendicular
}

export interface SegmentFrame {
  origin: [number, number, number]
  xAxis: [number, number, number]  // along wall
  yAxis: [number, number, number]  // outward normal (left of direction)
  zAxis: [number, number, number]  // up
}

/**
 * Convert wall centerline (Point2D[] in feet) into individual segments.
 * Each segment is a finite line between consecutive points.
 */
export function centerlineToSegments(centerline: Point2D[]): WallSegment[] {
  if (centerline.length < 2) return []

  const segments: WallSegment[] = []
  for (let i = 0; i < centerline.length - 1; i++) {
    const start = centerline[i]!
    const end = centerline[i + 1]!
    const dx = end.x - start.x
    const dy = end.y - start.y
    const length = Math.sqrt(dx * dx + dy * dy)
    const direction = { x: dx / length, y: dy / length }
    // Left-hand normal (rotate 90° counter-clockwise)
    const normal = { x: -dy / length, y: dx / length }

    segments.push({ start, end, length, direction, normal })
  }
  return segments
}

/**
 * Build a 3D segment frame from a 2D wall segment.
 * origin = segment start at z=0
 * xAxis = along wall direction
 * yAxis = outward normal (left of direction) = thickness direction
 * zAxis = up
 */
export function buildSegmentFrame(segment: WallSegment, heightInches: number): SegmentFrame {
  // Convert feet to inches for world coordinates (1 foot = 12 inches)
  // But in FRMX, we use feet as world units — let's use feet for 3D too
  // So 1 foot in 2D = 1 unit in 3D
  const ox = segment.start.x
  const oy = segment.start.y
  const oz = 0

  return {
    origin: [ox, oy, oz],
    xAxis: [segment.direction.x, segment.direction.y, 0],
    yAxis: [segment.normal.x, segment.normal.y, 0],
    zAxis: [0, 0, 1],
  }
}

/**
 * Transform a point from wall-local coords to world coords.
 * distAlong: distance along wall in feet
 * offset: perpendicular offset in feet (positive = exterior side)
 * height: height in feet
 */
export function wallToWorld(
  frame: SegmentFrame,
  distAlong: number,
  offset: number,
  height: number
): [number, number, number] {
  const [ox, oy, oz] = frame.origin
  const [xx, xy, xz] = frame.xAxis
  const [yx, yy, yz] = frame.yAxis
  const [zx, zy, zz] = frame.zAxis

  return [
    ox + xx * distAlong + yx * offset + zx * height,
    oy + xy * distAlong + yy * offset + zy * height,
    oz + xz * distAlong + yz * offset + zz * height,
  ]
}

/**
 * Create a BoxGeometry (in feet) for a wall segment.
 * Length = segment.length
 * Height = wall height in feet
 * Depth = wall thickness in feet
 * Centered on the segment midpoint, at the wall's base z=0
 */
export function createWallBoxGeometry(
  segment: WallSegment,
  heightFeet: number,
  thicknessFeet: number
): { position: [number, number, number]; args: [number, number, number] } {
  // Midpoint of segment
  const mx = (segment.start.x + segment.end.x) / 2
  const my = (segment.start.y + segment.end.y) / 2
  const mz = heightFeet / 2  // center vertically

  // Rotation: face along the wall direction
  // The box width (X) = segment.length
  // box height (Y) = thickness (NOT the default — need rotation)
  // Actually three.js BoxGeometry args are [width, height, depth]
  // We want: width=segment.length, height=wallHeight, depth=thickness
  // No rotation needed if we orient the box along X axis

  return {
    position: [mx, my, mz],
    args: [segment.length, heightFeet, thicknessFeet],
  }
}
```

```typescript
// apps/web/src/features/three/utils/layerStackToOffsets.ts
import type { Layer } from '@frmx/core'

export interface LayerOffsets {
  exteriorOffset: number   // distance from centerline to exterior face (feet)
  interiorOffset: number   // distance from centerline to interior face (feet)
  totalThickness: number   // exteriorOffset + interiorOffset
}

/**
 * Compute exterior/interior offsets from a layer stack.
 * Sum exterior layers → exterior half-thickness
 * Sum interior layers → interior half-thickness
 */
export function layerStackToOffsets(layers: Layer[]): LayerOffsets {
  let exteriorSum = 0
  let interiorSum = 0

  for (const layer of layers) {
    const thicknessFeet = layer.thickness / 12  // convert inches to feet
    switch (layer.position) {
      case 'exterior':
        exteriorSum += thicknessFeet
        break
      case 'interior':
        interiorSum += thicknessFeet
        break
      case 'core':
        // Core layers (stud cavity) span between exterior and interior
        // Don't add to either offset — they fill the middle
        break
    }
  }

  return {
    exteriorOffset: exteriorSum,
    interiorOffset: interiorSum,
    totalThickness: exteriorSum + interiorSum,
  }
}

/**
 * Default wall thickness for common wall types (used when no layer stack available).
 * Returns thickness in feet.
 */
export function defaultWallThickness(wallType: string): number {
  switch (wallType) {
    case 'exterior':
      return 5.5 / 12  // 5.5 inches = 0.458 feet
    case 'interior partition':
      return 4.5 / 12  // 4.5 inches = 0.375 feet
    default:
      return 6 / 12    // 6 inches = 0.5 feet
  }
}
```

```typescript
// apps/web/src/features/three/hooks/useWalls3D.ts
import { useMemo } from 'react'
import { useFRMXStore } from '@/store'
import type { Wall } from '@frmx/core'
import { centerlineToSegments, type WallSegment } from '../utils/wallToGeometry'
import { layerStackToOffsets, defaultWallThickness } from '../utils/layerStackToOffsets'

export interface WallInstance3D {
  id: string
  name: string
  wallType: string
  height: number           // in feet
  thickness: number        // in feet (total)
  segments: WallSegment[]
  centerline: { x: number, y: number }[]
}

/**
 * Derive 3D wall instances from the Zustand store.
 * Each wall's centerline is converted to segments for 3D rendering.
 */
export function useWalls3D(): WallInstance3D[] {
  const levels = useFRMXStore(s => s.project.building.levels)

  const walls = useMemo(() => {
    const result: WallInstance3D[] = []

    for (const level of levels) {
      for (const wall of level.walls) {
        // Convert wall height from inches to feet
        const heightFeet = wall.height / 12

        // Get wall thickness from layer stack or default
        const firstPanel = wall.panels[0]
        let thicknessFeet: number
        if (firstPanel?.layerStack && firstPanel.layerStack.length > 0) {
          const offsets = layerStackToOffsets(firstPanel.layerStack)
          thicknessFeet = offsets.totalThickness
        } else {
          thicknessFeet = defaultWallThickness(wall.wallType)
        }

        result.push({
          id: wall.id,
          name: wall.name,
          wallType: wall.wallType,
          height: heightFeet,
          thickness: thicknessFeet,
          segments: centerlineToSegments(wall.centerline),
          centerline: wall.centerline,
        })
      }
    }

    return result
  }, [levels])

  return walls
}
```

```typescript
// apps/web/src/features/three/components/Wall3D.tsx
'use client'

import { useMemo, useRef } from 'react'
import * as THREE from 'three'
import { useFrame } from '@react-three/fiber'
import { mergeGeometries } from 'three/examples/jsm/utils/BufferGeometryUtils.js'
import type { WallInstance3D } from '../hooks/useWalls3D'
import { STUD_MATERIAL, EXTERIOR_SHEATHING_MATERIAL } from '../utils/materials'

interface Wall3DProps {
  wall: WallInstance3D
  isSelected: boolean
  onSelect: (id: string) => void
}

export default function Wall3D({ wall, isSelected, onSelect }: Wall3DProps) {
  const meshRef = useRef<THREE.Group>(null)

  // Build a merged geometry for all segments of this wall
  const geometry = useMemo(() => {
    const geometries: THREE.BufferGeometry[] = []

    for (const segment of wall.segments) {
      // Create a box for each segment
      // Width = segment.length, Height = wall height, Depth = wall thickness
      const geo = new THREE.BoxGeometry(
        segment.length,
        wall.height,
        wall.thickness
      )

      // Position the box at the segment midpoint, centered vertically
      const mx = (segment.start.x + segment.end.x) / 2
      const my = (segment.start.y + segment.end.y) / 2
      const mz = wall.height / 2

      geo.translate(mx, my, mz)
      geometries.push(geo)
    }

    // Merge all segment geometries into one
    return mergeGeometries(geometries)
  }, [wall.segments, wall.height, wall.thickness])

  // Highlight on hover
  const material = useMemo(() => {
    return new THREE.MeshStandardMaterial({
      color: isSelected ? 0x4488ff : 0xd4c4a8,
      roughness: 0.8,
      metalness: 0,
    })
  }, [isSelected])

  return (
    <group
      ref={meshRef}
      onClick={(e) => {
        e.stopPropagation()
        onSelect(wall.id)
      }}
      onPointerEnter={(e) => {
        document.body.style.cursor = 'pointer'
      }}
      onPointerLeave={(e) => {
        document.body.style.cursor = 'default'
      }}
    >
      <mesh
        geometry={geometry}
        material={material}
        castShadow
        receiveShadow
      />

      {/* Render studs as separate geometry */}
      <Studs3D wall={wall} />

      {/* Selection outline */}
      {isSelected && <SelectionOutline wall={wall} />}
    </group>
  )
}

/**
 * Render studs for a wall based on the framing model.
 * For now, simplified studs at regular intervals.
 */
function Studs3D({ wall }: { wall: WallInstance3D }) {
  const studGeometry = useMemo(() => {
    // 2x4 stud: 1.5" x 3.5" (in feet)
    const w = 1.5 / 12
    const d = 3.5 / 12
    return new THREE.BoxGeometry(w, wall.height, d)
  }, [wall.height])

  const studs = useMemo(() => {
    const result: { position: [number, number, number] }[] = []
    const spacing = 16 / 12  // 16 inches = 1.33 feet

    for (const segment of wall.segments) {
      const numStuds = Math.floor(segment.length / spacing)
      for (let i = 0; i <= numStuds; i++) {
        const t = i / Math.max(numStuds, 1)
        const x = segment.start.x + t * (segment.end.x - segment.start.x)
        const y = segment.start.y + t * (segment.end.y - segment.start.y)
        const z = wall.height / 2
        result.push({ position: [x, y, z] })
      }
    }
    return result
  }, [wall.segments, wall.height])

  return (
    <>
      {studs.map((stud, i) => (
        <mesh
          key={i}
          geometry={studGeometry}
          material={STUD_MATERIAL}
          position={stud.position}
          castShadow
        />
      ))}
    </>
  )
}
```

```typescript
// apps/web/src/features/three/components/Room3D.tsx
'use client'

import { useMemo } from 'react'
import * as THREE from 'three'
import type { WallInstance3D } from '../hooks/useWalls3D'

interface Room3DProps {
  walls: WallInstance3D[]
}

/**
 * Generate floor polygon from wall centerlines.
 * Computes the bounding polygon of all walls to create a room floor.
 */
export default function Room3D({ walls }: Room3DProps) {
  const floorGeometry = useMemo(() => {
    if (walls.length === 0) return null

    // Collect all wall centerline points
    const points: { x: number; y: number }[] = []
    for (const wall of walls) {
      for (const pt of wall.centerline) {
        points.push(pt)
      }
    }

    if (points.length < 3) return null

    // Compute bounding box as simple floor approximation
    // For proper room detection, we'd need polygon triangulation
    const xs = points.map(p => p.x)
    const ys = points.map(p => p.y)
    const minX = Math.min(...xs)
    const maxX = Math.max(...xs)
    const minY = Math.min(...ys)
    const maxY = Math.max(...ys)

    // Create a simple quad floor
    const geo = new THREE.PlaneGeometry(maxX - minX, maxY - minY)
    geo.translate((minX + maxX) / 2, (minY + maxY) / 2, 0)
    return geo
  }, [walls])

  if (!floorGeometry) return null

  return (
    <mesh geometry={floorGeometry} rotation={[-Math.PI / 2, 0, 0]} receiveShadow>
      <meshStandardMaterial color={0x8b7355} roughness={0.6} />
    </mesh>
  )
}
```

```typescript
    target: [0, 0, 0],
    fov: 50,
    orthographic: false,
  },
  walkthrough: {
    position: [0, 5, 10],
    target: [0, 5, 0],
    fov: 75,
    orthographic: false,
  },
}

export function useCameraPresets(
  controlsRef: React.MutableRefObject<OrbitControlsType | null>
) {
  const applyPreset = useCallback((preset: CameraPreset) => {
    const controls = controlsRef.current
    if (!controls) return

    const state = PRESETS[preset]
    controls.object.position.set(...state.position)
    controls.target.set(...state.target)
    controls.update()
  }, [controlsRef])

  return { applyPreset, presets: PRESETS }
}
```

```typescript
import type { OrbitControls as OrbitControlsImpl } from 'three-stdlib'

export function setup3DKeyboardShortcuts(
  controlsRef: React.MutableRefObject<OrbitControlsImpl | null>
) {
  const handleKeyDown = (e: KeyboardEvent) => {
    const controls = controlsRef.current
    if (!controls) return

    switch (e.code) {
      case 'Space':
        // Toggle between orbit and first-person
        e.preventDefault()
        // Could implement walk mode toggle here
        break
      case 'KeyF':
        // Focus on selected
        e.preventDefault()
        // Zoom to fit selected wall
        break
      case 'KeyR':
        // Reset camera
        e.preventDefault()
        controls.object.position.set(30, 30, 30)
        controls.target.set(0, 0, 0)
        controls.update()
        break
    }
  }

  window.addEventListener('keydown', handleKeyDown)
  return () => window.removeEventListener('keydown', handleKeyDown)
}
```

## Implementation Notes

- **React 19 Compatibility:** @react-three/fiber 8.x has known issues with React 19. If you encounter errors, check the R3F GitHub issues for workarounds (typically adding `typeof window !== 'undefined'` guards or using `createRoot` instead of `hydrateRoot`).
- **Wall thickness** is derived from layer stacks. For walls without panels, `defaultWallThickness()` uses wall type to estimate (exterior: 5.5", interior partition: 4.5").
- **Corner mitering** is simplified — walls are rendered as separate boxes that may have small gaps at corners. True CSG-based wall joining is an advanced feature.
- **Stud rendering** is simplified and shows studs at regular 16" intervals along each segment. Full framing integration with the framing-engine package comes later.
- **Room detection** uses a simple bounding box. Proper room boundaries from wall polygons require a polygon triangulation library.
